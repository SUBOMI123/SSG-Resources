import { prisma } from './db';
import { InventoryUpdateRequest, InventoryUpdateResponse, InventoryItem, InventoryError } from '@/types/inventory';

export async function getInventory(product_id: string): Promise<InventoryItem | null> {
  return prisma.product.findUnique({
    where: { id: product_id },
  }) as Promise<InventoryItem | null>;
}

export async function listInventory(): Promise<InventoryItem[]> {
  return prisma.product.findMany({
    orderBy: { created_at: 'asc' },
  }) as Promise<InventoryItem[]>;
}

export async function updateInventory(
  request: InventoryUpdateRequest
): Promise<InventoryUpdateResponse | InventoryError> {
  try {
    // Per D-02: Use atomic transaction to prevent race conditions
    const result = await prisma.$transaction(async (tx) => {
      // 1. Fetch current product with lock
      const product = await tx.product.findUnique({
        where: { id: request.product_id },
      });

      if (!product) {
        throw new Error('PRODUCT_NOT_FOUND');
      }

      if (request.new_quantity < 0) {
        throw new Error('INVALID_QUANTITY');
      }

      const previous_quantity = product.quantity_on_hand;
      const change_amount = request.new_quantity - previous_quantity;

      // 2. Update product quantity
      const updated = await tx.product.update({
        where: { id: request.product_id },
        data: { quantity_on_hand: request.new_quantity },
      });

      // 3. Log to audit trail per D-07
      const history = await tx.inventoryHistory.create({
        data: {
          product_id: request.product_id,
          previous_quantity,
          new_quantity: request.new_quantity,
          change_reason: request.reason,
          changed_by: request.changed_by,
        },
      });

      return {
        product: updated,
        previous_quantity,
        change_amount,
        history_id: history.id,
      };
    });

    return {
      success: true,
      ...result,
    };
  } catch (error) {
    const message = String(error);
    if (message.includes('PRODUCT_NOT_FOUND')) {
      return { code: 'PRODUCT_NOT_FOUND', message: 'Product not found' };
    }
    if (message.includes('INVALID_QUANTITY')) {
      return { code: 'INVALID_QUANTITY', message: 'Quantity cannot be negative' };
    }
    return { code: 'TRANSACTION_FAILED', message: 'Failed to update inventory' };
  }
}

export async function reserveInventory(
  product_id: string,
  quantity: number
): Promise<boolean> {
  // For order placement: reserve stock without deducting
  try {
    await prisma.product.update({
      where: { id: product_id },
      data: { reserved: { increment: quantity } },
    });
    return true;
  } catch {
    return false;
  }
}

export async function releaseReservation(
  product_id: string,
  quantity: number
): Promise<boolean> {
  try {
    await prisma.product.update({
      where: { id: product_id },
      data: { reserved: { decrement: quantity } },
    });
    return true;
  } catch {
    return false;
  }
}

export async function getInventoryHistory(product_id: string, limit = 50) {
  return prisma.inventoryHistory.findMany({
    where: { product_id },
    orderBy: { created_at: 'desc' },
    take: limit,
  });
}
```

import { NextRequest, NextResponse } from 'next/server';
import { listInventory, updateInventory } from '@/lib/inventory';

export async function GET() {
  try {
    const inventory = await listInventory();
    return NextResponse.json({ success: true, data: inventory });
  } catch (error) {
    return NextResponse.json({ error: 'Failed to fetch inventory' }, { status: 500 });
  }
}

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const result = await updateInventory({
      product_id: body.product_id,
      new_quantity: body.new_quantity,
      reason: body.reason || 'MANUAL_UPDATE',
      changed_by: body.changed_by || 'admin',
    });

    if ('code' in result) {
      return NextResponse.json(result, { status: 400 });
    }

    return NextResponse.json({ success: true, data: result });
  } catch (error) {
    return NextResponse.json({ error: 'Invalid request' }, { status: 400 });
  }
}
```
